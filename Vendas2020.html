# criando uma funo na qual faa o comprimento de caracteres na coluna 'Cpf/Cnpj'.
# em seguida criei uma coluna informando o total desses caracteres.
# sei que os registros de CPF tem o total de 14 caracteres, e CNPJ de 18 caracteres.

def tipo_cliente(x): # nome da funo, junto com uma varivel nela
    return len(x) # retorno da funo dentro de uma outra funo

tipo_cliente(vendas['cpf_cnpj'])

vendas['tipo_cliente'] = vendas['cpf_cnpj'].apply(tipo_cliente)

# criando uma funo na qual substitua os valores 14 e 18 pelos nomes corretos de 'Pessoa fsica' e 'Pessoa jurdica'.
def tipo(num): 
    if num == 14: # estrutura if para buscar a varivel igual a 14 caracteres      
        return 'pessoa_fsica'
    else: # se no, buscar a varivel com 18 caracteres
        num == 18
        return 'pessoa_jurdica'

vendas['tipo_cliente'] = vendas['tipo_cliente'].map(tipo) # resolvi criar uma coluna
